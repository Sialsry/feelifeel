## 자바스크립트 변수

> [예약어] [변수명];

```js
let num; //변수 선언
num = 5; //변수값 할당
//5의 값의 주소는 재할당 이후에 가비지컬렉터가 해제시킨다.
num = 6; //변수값 재할당
//새로운 주소를 num에게 참조해준다 6의 값을 가리키는
//자바스크립트 원시값의 불변성

let num;//초기화 되지 않음. null 값
//let은 재선언 안됨

var num2;//초기에 undefined 값을 임시로 넣어준다. 
//var는 재선언과 재할당 모두 가능

const num3;
//let과 같지만 재선언 뿐만 아니라 재할당도 안됨

//es6에서 let, const 목적성 첫번째 호이스팅 


var num2;
//원시타입종류, TDZ, 호이스팅(var, let 차이)
```

### 연산자

```js
// 단항 산술 연산자
let num = 1;
let num2 = 2;
let num3 = num + num2;
// 문자열과 숫자를 더하는 경우
// "1" + 1 => "11"

// 비교 연산자

"4" == 4; // 값만 비교를 하기 때문에 자료형은 신경쓰지 않는다
"true" == true;
// 데이터의 타입까지 비교하는 연산자 : ===



// 부정 연산자: bool 타입의 값을 반전시킬때, 변수의 값을 확인할때
console.log(!true)


console.log(4 != 4); // 비교해서 값이 다르면 true
console.log(4 !== 4); // 비교해서 값 뿐만 아니라 타입까지 다르면 true

// 큰 값과 작은 값을 비교하는 연산자 



```

### 반복문
> 반복적으로 실행되어야 하는 코드를 작성할 때
```js
[for 예약어] (선언문; 조건식; 증감식) {

}

for (let i = 0; i < 10; i++) {
    for (let j = 0; j < 10; j++) {
    console.log(i, j);
}
}

```
